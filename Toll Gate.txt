
#include <Servo.h>

//=========================================================
// Pin Numbers 
//=========================================================
const int TRIG_PIN = 9;
const int ECHO_PIN = 10;
const int SERVO_PIN = 3;

//=========================================================
// Adjustable Settings
//=========================================================
const int DETECT_DISTANCE = 15;   // Detect object within 15 cm
const int GATE_OPEN_ANGLE = 90;   // Servo angle when gate opens
const int GATE_CLOSE_ANGLE = 0;   // Servo angle when gate closes
const int GATE_OPEN_TIME = 2000;  // Keep gate open for 2 seconds

Servo gateServo;

//=========================================================
// Setup
//=========================================================
void setup()
{
  Serial.begin(9600);

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  gateServo.attach(SERVO_PIN);

  // Gate starts closed
  gateServo.write(GATE_CLOSE_ANGLE);
}

//=========================================================
// Main Loop
//=========================================================
void loop()
{
  // Read distance from ultrasonic sensor
  int distance = getDistance();

  Serial.print("Distance : ");
  Serial.print(distance);
  Serial.println(" cm");

  // Check if object is detected
  if (distance > 0 && distance <= DETECT_DISTANCE)
  {
    Serial.println("Object Detected!");

    // Open gate
    gateServo.write(GATE_OPEN_ANGLE);

    // Keep gate open
    delay(GATE_OPEN_TIME);

    // Close gate
    gateServo.write(GATE_CLOSE_ANGLE);

    // Small delay to avoid repeated triggering
    delay(500);
  }

  delay(100);
}

//=========================================================
// Function to Measure Distance
//=========================================================
int getDistance()
{
  long duration;
  int distance;

  // Send ultrasonic pulse
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);

  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);

  digitalWrite(TRIG_PIN, LOW);

  // Read echo time
  duration = pulseIn(ECHO_PIN, HIGH);

  // Convert time to distance
  distance = duration * 0.034 / 2;

  return distance;
}
