/*
=========================================================
Project Name : Automatic Door Opener
Board        : Arduino UNO

#include <Servo.h>

//=========================================================
// Pin Numbers 
//=========================================================
const int TRIG_PIN = 9;
const int ECHO_PIN = 10;
const int SERVO_PIN = 3;

//=========================================================
// Adjustable Settings
//=========================================================
const int DETECT_DISTANCE = 20;   // Detection distance (cm)

const int DOOR_OPEN = 90;         // Door open angle
const int DOOR_CLOSE = 0;         // Door close angle

const int CLOSE_DELAY = 2000;     // Close after 2 seconds

Servo doorServo;

bool doorOpen = false;
unsigned long objectGoneTime = 0;

//=========================================================
// Setup
//=========================================================
void setup()
{
  Serial.begin(9600);

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  doorServo.attach(SERVO_PIN);

  // Door starts closed
  doorServo.write(DOOR_CLOSE);
}

//=========================================================
// Main Loop
//=========================================================
void loop()
{
  int distance = getDistance();

  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");

  //=======================================================
  // Object Detected
  //=======================================================
  if (distance > 0 && distance <= DETECT_DISTANCE)
  {
    // Open the door
    doorServo.write(DOOR_OPEN);

    doorOpen = true;

    // Reset timer while object is present
    objectGoneTime = millis();
  }

  //=======================================================
  // No Object Detected
  //=======================================================
  else
  {
    if (doorOpen)
    {
      // Close door after 2 seconds
      if (millis() - objectGoneTime >= CLOSE_DELAY)
      {
        doorServo.write(DOOR_CLOSE);
        doorOpen = false;
      }
    }
  }

  delay(100);
}

//=========================================================
// Function to Measure Distance
//=========================================================
int getDistance()
{
  long duration;

  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);

  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);

  digitalWrite(TRIG_PIN, LOW);

  duration = pulseIn(ECHO_PIN, HIGH);

  int distance = duration * 0.034 / 2;

  return distance;
}
