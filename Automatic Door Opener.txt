#include <Servo.h>

//=========================================================
// Pin Numbers
//=========================================================
const int TRIG_PIN = 9;
const int ECHO_PIN = 10;
const int SERVO_PIN = 3;

//=========================================================
// Adjustable Settings
//=========================================================
const int DETECT_DISTANCE = 20;   // cm
const int DOOR_OPEN = 90;         // Servo angle
const int DOOR_CLOSE = 0;
const unsigned long CLOSE_DELAY = 2000; // ms

Servo doorServo;

bool doorOpen = false;
unsigned long objectGoneTime = 0;

//=========================================================
// Setup
//=========================================================
void setup()
{
  Serial.begin(9600);

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  doorServo.attach(SERVO_PIN);
  doorServo.write(DOOR_CLOSE);

  delay(500);
}

//=========================================================
// Main Loop
//=========================================================
void loop()
{
  int distance = getDistance();

  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");

  // Object detected
  if (distance != -1 && distance <= DETECT_DISTANCE)
  {
    if (!doorOpen)
    {
      doorServo.write(DOOR_OPEN);
      doorOpen = true;
      Serial.println("Door Open");
    }

    objectGoneTime = millis();
  }

  // No object detected
  else
  {
    if (doorOpen && (millis() - objectGoneTime >= CLOSE_DELAY))
    {
      doorServo.write(DOOR_CLOSE);
      doorOpen = false;
      Serial.println("Door Closed");
    }
  }

  delay(100);
}

//=========================================================
// Function to Measure Distance
//=========================================================
int getDistance()
{
  // Clear trigger
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);

  // Send 10 µs pulse
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  // Read echo with timeout (30 ms ≈ 5 m)
  long duration = pulseIn(ECHO_PIN, HIGH, 30000);

  // No echo received
  if (duration == 0)
  {
    return -1;
  }

  // Convert to centimeters
  int distance = duration * 0.0343 / 2;

  return distance;
}
